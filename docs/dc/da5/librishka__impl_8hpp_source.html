195static inline string get_rt_string(u32 len) {
196 string str;
197 Memory::set(str, 0, len + 1);
198
199 for(u32 i = 0; i < len; i++)
200 str[i] = rt_strpass();
201
202 str[len] = '\0';
203 return str;
204}
